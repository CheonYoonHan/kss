        function GoodsImagePath(goodsNo) {

    function SetAddContent(data, rpType)
    {
        var ret = "",
            splitData;

        if(data.length > 0)
        {
            splitData = data.split('＃');
            if(Number(splitData[0]) == rpType)

                    }
                }
                else
                {
                    var ifwidth = (Number(splitData[3]) > 780) ? "780":splitData[3];
                    //링크 타입 처리
                    if(splitData[2].indexOf("youtube.com") > -1)
                    {
                        ret = "<br/><br/> <iframe src='//www.youtube.com/embed/" + splitData[2].split('?v=')[splitData[2].split('?v=').length - 1] + "' width='"+ifwidth+"' height='"+splitData[4]+"'  frameborder='0' allowfullscreen='' ></iframe>";
                    }
                    else if(splitData[2].indexOf("youtu.be") > -1)
                    {
                        ret = "<br/><br/> <iframe src='//www.youtube.com/embed/" + splitData[2].split('/')[splitData[2].split('/').length - 1] + "' width='"+ifwidth+"' height='"+splitData[4]+"'  frameborder='0' allowfullscreen='' ></iframe>";
                    }
                    else if(splitData[2].indexOf("vimeo.com") > -1)
                    {
                        ret = "<br/><br/> <iframe src='//player.vimeo.com/video/" + splitData[2].split('/')[splitData[2].split('/').length - 1] + "' width='"+ifwidth+"' height='"+splitData[4]+"'  frameborder='0' allowfullscreen='' ></iframe>";

    function f_GoodsLimitedShowTime(data){

        var LimitedStart = eval(data.result.LimitedStart.replace(/\/Date\((\d+)\)\//gi, "new Date($1)"));
        var LimitedEnd = eval(data.result.LimitedEnd.replace(/\/Date\((\d+)\)\//gi, "new Date($1)"));
        var NowStamp = eval(data.result.NowStamp.replace(/\/Date\((\d+)\)\//gi, "new Date($1)"));

        NowStamp = new Date(NowStamp .getTime() + 1000 * limitedTimerCounter);

        var difference = LimitedEnd.getTime() - NowStamp.getTime();

        if (difference <= 0) {

            alert("리미티드 상품 판매가 마감되었습니다.");
            $(location).attr('href',"/");
            clearInterval(limitedTimer);

        } else {

            var imgNum = 12;

            $.each(data.dateList, function (key, data) {

                var StepTime = eval(data.replace(/\/Date\((\d+)\)\//gi, "new Date($1)"));
                var gap = StepTime.getTime() - NowStamp.getTime();

                if (gap <= 0) {
                    imgNum--;
                }
            })

            var seconds = Math.floor(difference / 1000);
            var minutes = Math.floor(seconds / 60);
            var hours = Math.floor(minutes / 60);
            var days = Math.floor(hours / 24);

            var result = "";

            hours %= 24;
            minutes %= 60;
            seconds %= 60;

            if(imgNum < 1) imgNum = 1;

            if(days > 0){
                //result += days + " 일 ";
                hours = (days * 24) + hours;
            }

            hours = f_TimerDigit(hours);
            minutes = f_TimerDigit(minutes);
            seconds = f_TimerDigit(seconds);
            imgNum = f_TimerDigit(imgNum);

            //if(days > 0){
            //    result += days + " 일 ";
            //    //hours = (days * 24) + hours;
            //}


            result += hours + ":" + minutes + ":" + seconds;

            var imgName = "https://funshop1.akamaized.net/www/common/limited_"+imgNum+".png";

            if(limitedTimerCounter == 0){

                var baseElement = $(".name");
                var pos = baseElement.position();

                var posTop = baseElement.offset().top;
                var posLeft = baseElement.offset().left;
                var width = $("#divIsLimitedArea").outerWidth();
                var height = $("#divIsLimitedArea").outerHeight();

                //var divIsLimitedTop= posTop - height/2 + 30;
                //var divIsLimitedLeft= posLeft - height- 15;

                var divIsLimitedTop= -50;
                var divIsLimitedLeft= -200;

                if(data.result.LimitedMemo != "" && data.result.LimitedMemo != null){
                    $("#limitedTimeText").text(data.result.LimitedMemo);
                    $("#divIsLimitedText").show();
                }

                $("#divIsLimitedArea").css({
                    position: "fixed",
                    width: width,
                    height: height
                }).show();

                $("#divIsLimited").css({
                    backgroundImage:'url('+imgName+')',
                    backgroundRepeat: "no-repeat"
                }).show();

                //$("#divIsLimited").css({
                //    position: "absolute",
                //    top: (divIsLimitedTop) + "px",
                //    left: (divIsLimitedLeft) + "px",
                //    backgroundImage:'url('+imgName+')',
                //    width: width,
                //    height: height
                //}).show();
            }else{
                $("#divIsLimited").css({backgroundImage:'url('+imgName+')'});
            }

            $("#LimitedTime").text(result);

            limitedTimerCounter++;
        }
    }

    }     function copyToClipboard(elem) {
        // create hidden text element, if it doesn't already exist
        var targetId = "_hiddenCopyText_";
        var isInput = elem.tagName === "INPUT" || elem.tagName === "TEXTAREA";
        var origSelectionStart, origSelectionEnd;
        if (isInput) {
            // can just use the original source element for the selection and copy
            target = elem;
            origSelectionStart = elem.selectionStart;
            origSelectionEnd = elem.selectionEnd;
        } else {
            // must use a temporary form element for the selection and copy
            target = document.getElementById(targetId);
            if (!target) {
                var target = document.createElement("textarea");
                target.style.position = "absolute";
                target.style.left = "-9999px";
                target.style.top = "0";
                target.id = targetId;
                document.body.appendChild(target);
            }
            target.textContent = elem.textContent;
        }
        // select the content
        var currentFocus = document.activeElement;
        target.focus();
        target.setSelectionRange(0, target.value.length);

        // copy the selection
        var succeed;
        try {
            succeed = document.execCommand("copy");
        } catch(e) {
            succeed = false;
        }
        // restore original focus
        if (currentFocus && typeof currentFocus.focus === "function") {
            currentFocus.focus();
        }

        if (isInput) {
            // restore prior selection
            elem.setSelectionRange(origSelectionStart, origSelectionEnd);
        } else {
            // clear temporary content
            target.textContent = "";
        }
        return succeed;
    }

    function SetCommentScore(score) {
        var returnVal = "";
        if (score > 0) {
            var maxvalue = Math.floor(score / 2)
                returnVal = "<span class='star'>";
                for (var i = 1; i <= 5; i++) {
                    if (i <= maxvalue) {
                        returnVal += "<img src='https://funshop.akamaized.net/www/detail/star_on.png' alt=''>";
                    }
                    else if ((score % 2 != 0) && (i == maxvalue + 1)) {
                        returnVal += "<img src='https://funshop.akamaized.net/www/detail/star_half.png' alt=''>";
                    }
                    else {
                        returnVal += "<img src='https://funshop.akamaized.net/www/detail/star_off.png' alt=''>";
                    }
                }
                returnVal += "</span>"
            }
            return returnVal;
        }
